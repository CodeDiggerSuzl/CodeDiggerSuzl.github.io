func doWork(id int,
	w worker) {
	for n := range w.in {
		fmt.Printf("Worker %d received %c\n",
			id, n)
		w.done()
	}
}

type worker struct {
	in   chan int
	done func() // 函数式方程 使用 func 进行 包装
}

func createWorker(
	id int, wg *sync.WaitGroup) worker {
	w := worker{
		in: make(chan int),
		done: func() { // 初始化 worker 的时候来对 done 实现
			wg.Done()
		},
	}
	go doWork(id, w)
	return w
}

func chanDemo() {
	// 生成 WaitGroup
	var wg sync.WaitGroup

	var workers [10]worker
	for i := 0; i < 10; i++ {
		workers[i] = createWorker(i, &wg)
	}
	// 20 个任务
	wg.Add(20)
	for i, worker := range workers {
		worker.in <- 'a' + i
		// 也可以每次 add(1)
		// wg.Add(1)
	}
	for i, worker := range workers {
		worker.in <- 'A' + i
	}

	wg.Wait()
}

func generator() chan int {
	out := make(chan int)
	go func() {
		i := 0
		for {
			time.Sleep(
				// 随机 sleep
				time.Duration(rand.Intn(1500)) *
					time.Millisecond)
			out <- i
			i++
		}
	}()
	return out
}

func worker(id int, c chan int) {
	for n := range c {
		time.Sleep(time.Second)
		fmt.Printf("Worker %d received %d\n",
			id, n)
	}
} 

func createWorker(id int) chan<- int {
	c := make(chan int)
	go worker(id, c)
	return c
}

func main() {
	// 输入的 goroutine
	var c1, c2 = generator(), generator()
	//输出的 goroutine
	var worker = createWorker(0)

	var values []int
	
	tm := time.After(10 * time.Second)
	// 定时器
	tick := time.Tick(time.Second)
	// 总控的 goroutine
	for {
		var activeWorker chan<- int  // 制成 nil channel 当数据还没准备好的时候
		var activeValue int
		// 当 🈶值的时候
		if len(values) > 0 {
			activeWorker = worker
			activeValue = values[0]
		}

		select {
		case n := <-c1:
			// 对数据进行排队
			values = append(values, n)
		case n := <-c2:
			values = append(values, n)
		case activeWorker <- activeValue:
			// 将第一个值拿掉
			values = values[1:]
		// 计时器的使用 800 毫秒,还进行800 毫米没生成数据
		case <-time.After(800 * time.Millisecond):
			fmt.Println("timeout")
		case <-tick:
			fmt.Println(
				"queue len =", len(values))
		case <-tm:
			fmt.Println("bye")
			return
		}
	}
}

// 用的时候还是使用 系统的原子类
type atomicInt struct{
    value int
    lock sync.Mutex
}

func (a *atomicInt) increment{
    a.lock.Lock()
    defer a.lock.Unlock()
    a.value++
    // 当想使用一段代码块中进行锁的时候 使用匿名函数
    func (){
    a.lock.Lock()
    defer a.lock.Unlock()
    a.value++
    }()
}

传统同步机制很少使用 尽量使用 channel

总结

协程

  • 轻量级线程
  • 非抢占式的多任务处理,由协程主动交出控制权
  • 在编译器/虚拟机/解释器层面上的多任务
  • 多个协可以在一个或多个线程运行
  • goroutine 和物理线程不是一一对应的
  • goroutine 是 map 到物理线程中的
  • 物理线程是一种抢占式的任务处理
