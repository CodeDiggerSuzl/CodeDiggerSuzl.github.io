}

总结

  1. 创建使用 make make (map[string] int)
  2. 获取元素 m[key]
  3. key 不存在的时候,会得到 value 的初始值
  4. 使用value,ok := m[key] 来判断是否存在 key
  5. 使用 delete 来删除一个 key
  6. 使用 range来遍历 key , 或者遍历 key value 对
  7. 不保证遍历顺序, 无需遍历,需要手动对 key 进行 排序.(将所有的key 放到 slice 中,使用 slice 可以排序)
  8. 使用 len()来获得长度
map 中的 key
  1. map 使用 hash 表,必须可以比较相同
  2. slice map function 不可以作为 key
  3. Struct 类型不包含上述类型, 也可以作为 key(编译时检查)

func getLength(s string) int {
	// 最出现的位置
	lastOccerd := make(map[byte]int)
	start := 0
	maxLength := 0

	//遍历 string
	for i, char := range []byte(s) {
		// if 中可以定义长度
		if lastI, ok := lastOccerd[char]; ok && lastI >= start {
			start = lastI + 1
		}
		if i-start+1 > maxLength {
			maxLength = i - start + 1
		}
		lastOccerd[char] = i
	}
	return maxLength
}
