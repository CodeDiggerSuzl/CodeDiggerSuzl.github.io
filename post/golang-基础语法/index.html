    func grade(score int) string {
	// 不用 switch score 直接 switch
	g := ""
	switch {
	case score > 100 || score < 0:
		panic(fmt.Sprintf("worng score :%d", score))
	case score < 60:
		g = "F"
	case score < 80:
		g = "D"
	default:
	}
	return g
}

func convert2Binary(n int) string {
	result := ""
	for ; n > 0; n /= 2 {
		lsb := n % 2
		result = strconv.Itoa(lsb) + result
	}
	return result
}

    package main

import (
	"fmt"
	"math"
	"reflect"
	"runtime"
)

//重写 pow
func pow(a, b int) int {
	return int(math.Pow(float64(a), float64(b)))

}

// 函数一公民 可当做参数
// 接受一个参数类型为 int int 返回值类为int 的函数 和两个int类型的参数 返回为int
func apply(op func(int, int) int, a, b int) int {
	// 使用 reflect 包来得到 op 的指针
	p := reflect.ValueOf(op).Pointer()
	opName := runtime.FuncForPC(p).Name()
	fmt.Println(opName, a, b)
	return op(a, b)
}
func main() {
	// 调用其他方法
	 fmt.Println(apply(pow,3,4))
	// 可直接写匿名函数 在调用的时候
	fmt.Println(apply(func(a int, b int) int {
		return int(math.Pow(float64(a), float64(b)))
	}, 5, 6))
}
