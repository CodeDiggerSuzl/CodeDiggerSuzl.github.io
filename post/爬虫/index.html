架构:

  1. 种子(起始 url)放入 engine
  2. engine 放入任务队列,从任务队列中拿任务 request
  3. engine 将 url 放入 fetcher 中返回 文本 text

import (
	"github.com/CodeDiggerSuzl/GoogleGolang/crawler/fetcher"
	"log"
)

// 可变长参数
func Run(seeds ...Request) {
	var requests []Request
	for _, r := range seeds {
		requests = append(requests, r)
	}
	for len(requests) > 0 {
		r := requests[0]
		requests = requests[1:]
		log.Printf("Fecthing %s", r.Url)
		body, err := fetcher.Fetch(r.Url)
		if err != nil {
			// 忽略这个
			log.Printf("Fecther:error fecthing url %s :%v>>>	 ", r.Url, err)
			continue
		}
		parseResult := r.ParserFunc(body)
		requests = append(requests, parseResult.Requests...) // ... 代表将 slice 中的 request 一个个展开加进去
		for _, item := range parseResult.Items {
			log.Printf("Got items %v", item)
		}
	}
}

import (
	"bufio"
	"fmt"
	"golang.org/x/net/html/charset"
	"golang.org/x/text/encoding"
	"golang.org/x/text/encoding/unicode"
	"golang.org/x/text/transform"
	"io/ioutil"
	"log"
	"net/http"
)

// 输入 url 返回
func Fetch(url string) ([]byte, error) {
	client := &http.Client{}
	req, err := http.NewRequest("GET", url, nil)
	if err != nil {
		log.Fatalln("NewRequest is err ", err)
		return nil, fmt.Errorf("NewRequest is err %v\n", err)
	}
	req.Header.Set("User-Agent", "Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/66.0.3359.181 Safari/537.36")
	resp, err := client.Do(req)
	if err != nil {
		return nil, fmt.Errorf("Error: http Get, err is %v\n", err)
	}

	defer resp.Body.Close()
	if resp.StatusCode != http.StatusOK {
		// error.NewError 2. fmt.Errorf
		return nil, fmt.Errorf("wrong status code: %d", resp.StatusCode)
	}
	bodyReader := bufio.NewReader(resp.Body)
	e := determineEncoding(bodyReader)
	utf8Reader := transform.NewReader(resp.Body, e.NewEncoder())
	return ioutil.ReadAll(utf8Reader)

}

// use x/text to guess encoding of a  file
func determineEncoding(r *bufio.Reader) encoding.Encoding {
	// use peek to get bytes
	bytes, err := bufio.NewReader(r).Peek(1024)
	if err != nil {
		log.Printf("Fetcher error :%v", err)
		// return default encoding
		return unicode.UTF8
	}
	e, _, _ := charset.DetermineEncoding(bytes, "")
	return e
}

const cityRe = `<a href="(http://album.zhenai.com/u/[0-9]+)" [^>]*>([^<]+)</a>`

const cityAllRe = `href="(http://album.zhenai.com/u/[0-9]+)" [^>]*>([^<]+)</a></th></tr> <tr><td[^>]*><span[^>]*>性别：</span>([^<]+)</td> <td><span[^>]*">居住地：</span>([^<]+)</td></tr> <tr><td[^>]*><span[^>]*>年龄：</span>([^<]*)</td> <td><span[^>]*>`

// 1 url
// 2. name
// 3. gender
// 3. location
// 4. age

func ParseCity(contents []byte) engine.ParseResult {
	re := regexp.MustCompile(cityAllRe)
	// matches := re.FindAll(contents, -1)
	matches := re.FindAllSubmatch(contents, -1)
	//每一个 request 要生成一个新的 url
	result := engine.ParseResult{}
	for _, m := range matches {
		name := string(m[2])
		profile := model.Profile{}
		profile.Name = name
		profile.Gender = string(m[3])
		profile.WorkLocation = string(m[4])
		profile.Age = string(m[5])
		result.Items = append(result.Items, "ser "+name)
		result.Requests = append(result.Requests, engine.Request{
			Url: string(m[1]),
			//ParserFunc: ParseProfile, // 类型不对了 函数式编程
			ParserFunc: func(contents []byte) engine.ParseResult {
				return ParseProfile(contents, profile) //太巧妙了  但是会导致名字
			},
		})
	}
	return result
}

import (

	"regexp"
)

const cityListRe = `<a href="(http://www.zhenai.com/zhenghun/[a-z0-9]+)"[^>]*>([^<]+)</a>`

// 返回解析结果
func ParseCityList(contents []byte) engine.ParseResult {
	// 获取 match 使用来进行获取 )
	re := regexp.MustCompile(cityListRe)
	// matches := re.FindAll(contents, -1)
	matches := re.FindAllSubmatch(contents, -1)
	//每一个 request 要生成一个新的 url
	result := engine.ParseResult{}
	limit := 4
	for _, m := range matches {
		result.Items = append(result.Items, "City "+string(m[2]))
		result.Requests = append(result.Requests, engine.Request{
			Url:        string(m[1]),
			ParserFunc: ParseCity,
		})
		limit--
		if limit == 0 {
			break
		}
	}
	return result
}

var constellationRe = regexp.MustCompile(`<div class="m-btn purple" data-v-bff6f798>([^<]+)(10.23-11.21)</div>`)
var heightRe = regexp.MustCompile(`div class="m-btn purple" data-v-bff6f798>([\d+]cm)</div>`)
var weightRe = regexp.MustCompile(`<div class="m-btn purple" data-v-bff6f798>([\d+]kg)</div>`)

//var workLocationRe = regexp.MustCompile(`<div class="m-btn purple" data-v-bff6f798>工作地:([^<]+)</div>`)
var incomeRe = regexp.MustCompile(`<div class="m-btn purple" data-v-bff6f798>月收入:([^<]+)</div>`)

//var OccupationRe = regexp.MustCompile(`<div class="m-btn purple" data-v-bff6f798>([^<]+)</div>`) 没有办法区分
//var ageRe = regexp.MustCompile(`<div class="m-btn purple" data-v-bff6f798>([\d+]岁)</div>`) // \d+ 一个或者多个数字
var marriageRe = regexp.MustCompile(`<div class="m-btn purple" data-v-bff6f798>([^<]+)</div>`)

func ParseProfile(contents []byte, profile model.Profile) engine.ParseResult { // 加入 name string 来进行name 值的传入

	//profile.Age = extractString(contents, ageRe)
	profile.Marriage = extractString(contents, marriageRe)
	profile.Height = extractString(contents, heightRe)
	profile.Weight = extractString(contents, weightRe)
	si := extractString(contents, constellationRe)
	fmt.Println(si)
	profile.Constellation = extractString(contents, constellationRe)
	//profile.WorkLocation = extractString(contents, workLocationRe)
	profile.Income = extractString(contents, incomeRe)
	result := engine.ParseResult{
		Items: []interface{}{profile},
	}
	return result
}

// 减少多次 regexp.MustCompile
func extractString(contents []byte, re *regexp.Regexp) string {
	match := re.FindSubmatch(contents)
	// 返回[][]byte 只要第一个
	if len(match) >= 2 {
		return string(match[1])
	} else {
		return ""
	}
}

import (
	"io/ioutil"
	"testing"
)

func TestParseCityList(t *testing.T) {
	contents, err := ioutil.ReadFile("cityList_test.html")
	if err != nil {
		panic(err)
	}
	// 测试机器上可能没有net 将测试文件保存在本地

	result := ParseCityList(contents)
	const resultSize = 470
	expectedUrls := []string{
		"http://www.zhenai.com/zhenghun/aba", "http://www.zhenai.com/zhenghun/akesu", "http://www.zhenai.com/zhenghun/alashanmeng",
	}
	expectedCitis := []string{
		"City 阿坝", "City 阿克苏", "City 阿拉善盟",
	}
	// verify the size
	if len(result.Requests) != resultSize {
		t.Errorf("request should have %d requests but only find %d", resultSize, len(result.Requests))
	}
	if len(result.Items) != resultSize {
		t.Errorf("request should have %d requests but only find %d", resultSize, len(result.Items))
	}
	//verify the result
	for i, url := range expectedUrls {
		if result.Requests[i].Url != url {
			t.Errorf("expected url #%d:%s ,but find %s ", i, url, result.Requests[i].Url)
		}
	}
	for i, city := range expectedCitis {
		if result.Items[i] != city {
			t.Errorf("expected url #%d:%s ,but find %s ", i, city, result.Items[i])
		}
	}
}
