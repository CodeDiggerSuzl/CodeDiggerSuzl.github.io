    // 是一个 任何类型的 的 slice
type Queue []interface{}
// 限定为 int
func (q *Queue) Push(v int) {
  //*q = append(*q, v)
  // 或者为 但是在运行的时候才会出错
  *q = append(*q,v.(int))
}
// 限制为 int
func (q *Queue) Pop() int {
  // TODO 为什么要加()
  head := (*q)[0]
  *q = (*q)[1:]
  return head.(int)// 将 interface{} 强制转换成 int
}
