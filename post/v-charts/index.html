<select id="getAvgIndexValue" resultType="java.util.LinkedHashMap">
    SELECT AVG(a.INDEX_SCORE) as value,
            b.INDEX_NAME      as indexName
    FROM cp_cr_results_detail a,
            cp_cr_index_sys_detail b
    WHERE a.INDEX_CODE = b.INDEX_CODE
        AND a.EVALUAT_DATE = #{year}
        AND b.INDEX_LEVEL = 1
    GROUP BY b.INDEX_NAME
</select>

     * 纵坐标

}

@Override
public ResultUtil test(String taskNo) {
    RadarAndInLineBo inLineBo = new RadarAndInLineBo();
    // {18,19,20}
    for (String s : years) {
        List<LinkedHashMap<String, Object>> avgIndexValue = cpCrResultsDetailExtMapper.getAvgIndexValue(s);
        LinkedHashMap<String, Object> rowElement = new LinkedHashMap<>();
        if (!avgIndexValue.isEmpty()) {
            inLineBo.columns.add("年份");
            for (LinkedHashMap<String, Object> map : avgIndexValue) {
                inLineBo.columns.add((String)map.get("indexName"));
                rowElement.put((String)map.get("indexName"), map.get("value"));
            }
            rowElement.put("年份", s);
            inLineBo.rows.add(rowElement);
        }
    }
    return ResultUtil.success(inLineBo);
