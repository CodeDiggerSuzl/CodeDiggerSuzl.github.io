		// handler 去做业务逻辑 出现错误  err,对 err 进行判断 实现统一处理
		err := handler(writer, request)

			code := http.StatusOK
			switch {
			case os.IsNotExist(err):
				code = http.StatusNotFound
       ....

panic

作用: 类似 throw

  1. 停止当前函数执行
  2. 一直向上返回,执行每一层的 defer
  3. 如果没有遇见 recover 程序就退出了

import (
	"fmt"

	"net/http"

const prefix = "/list/"
// TODO 🤢 自定义一种 userError 的类型 实现另一个 userError 接口()
type userError string
// 实现 userError 接口
func (e userError) Error() string {
	return e.Message()
}

func (e userError) Message() string {
	return string(e) // return e 自己 转换成 string
}
// 碰见 err就抛出
func HandleFileList(writer http.ResponseWriter,
	request *http.Request) error {
	fmt.Println()
	if strings.Index(
		request.URL.Path, prefix) != 0 {
		return userError(
			fmt.Sprintf("path %s must start "+
				"with %s",
				request.URL.Path, prefix))
	}
	path := request.URL.Path[len(prefix):]
	file, err := os.Open(path)
	if err != nil {
		return err
	}
	defer file.Close()

	all, err := ioutil.ReadAll(file)
	if err != nil {
		return err
	}

	writer.Write(all)
	return nil
}

package main

import (
	"log"
	"net/http"
	_ "net/http/pprof"
	"os"

	"imooc.com/ccmouse/learngo/errhandling/filelistingserver/filelisting"
)

// 定义一个 type 返回 error (连接 HandleFileList 和 errWrapper)
type appHandler func(writer http.ResponseWriter,
                     request *http.Request) error

// 函数式编程: 传入一个函数 返回一个函数对这个函数进行包装后返回 (并不是返回类型函数和传入函数的类型完全一样)
func errWrapper(handler appHandler) func(http.ResponseWriter,
                                         *http.Request) {
	//返回的 error
	return func(writer http.ResponseWriter,request *http.Request) {
		// panic
		defer func() {
			// 对 panic 进行保护
			// 使用匿名函数 能够调用 后面加上()
			if r := recover(); r != nil {
				log.Printf("Panic: %v", r)
				http.Error(writer,
					http.StatusText(http.StatusInternalServerError),
					http.StatusInternalServerError)
			}
		}()

		err := handler(writer, request)

		if err != nil {
			log.Printf("Error occurred "+
				"handling request: %s",
				err.Error())

			// user error  type assertion 区分 type 如果是 userError 就返回 自定义信息
			if userErr, ok := err.(userError); ok {
				http.Error(writer,
					userErr.Message(),
					http.StatusBadRequest)
				return  // 处理完了 return
			}

			// system error
			code := http.StatusOK
			switch {
			case os.IsNotExist(err):
				code = http.StatusNotFound
			case os.IsPermission(err):
				code = http.StatusForbidden
			default:
				code = http.StatusInternalServerError
			}
			http.Error(writer,
				http.StatusText(code), code)
		}
	}
}

// 自定义接口

type userError interface {
	error // 系统看
	Message() string //用户可以看到的错误
}

func main() {
	http.HandleFunc("/",errWrapper(filelisting.HandleFileList))

	err := http.ListenAndServe(":8888", nil)
	if err != nil {
		panic(err)
	}
}

error 和 panic

  1. 尽量不要用 panic
  2. 意料之中的要用 error 比如文件打不开
  3. 意料只为的使用 panic 比如数组越界
