package main

import (
	"fmt"
	"os"
)

func readMaze(filepath string) [][]int {
	file, err := os.Open(filepath)
	if err != nil {
		panic(err)
	}
	var row, col int
	fmt.Fscanf(file, "%d %d", &row, &col) // 读取一行 使用取地址才能改的了
	maze := make([][]int, row)            // row 指的是有多少行就是多少个 => []int
	for i := range maze {
		maze[i] = make([]int, col) // 每行有 5 列
		for j := range maze[i] {
			fmt.Fscanf(file, "%d", &maze[i][j])
		}
	}
	return maze
}

// point struct
type point struct {
	i, j int // 数组的 i j 和 x y 轴不同
}

// add two point
func (p point) add(r point) point {
	return point{p.i + r.i, p.j + r.j}
}

// if is out of the maze
func (p point) at(grid [][]int) (int, bool) {
	if p.i < 0 || p.i >= len(grid) {
		return 0, false
	}
	if p.j < 0 || p.j >= len(grid[p.i]) {
		return 0, false

	}
	return grid[p.i][p.j], true
}

// direction difnition
var dirs = [4]point{
	// up left down right
	{-1, 0}, {0, -1}, {1, 0}, {0, 1}}

func walk(maze [][]int, start, end point) [][]int {
	// steps :the routine just like maze
	steps := make([][]int, len(maze))
	for i := range steps {
		steps[i] = make([]int, len(maze[i]))
	}
	// put the start point into the queue
	Queue := []point{start}
	// when will end :1. when reach the maze end 2. when queue is empty

	for len(Queue) > 0 {
		// current
		current := Queue[0]
		Queue = Queue[1:]
		if current == end {
			break
		}
		for _, dir := range dirs {
			next := current.add(dir)
			// maze at next is 0
			// & steps at next is 0
			// & next != start
			val, ok := next.at(maze)
			if !ok || val == 1 {
				continue
			}
			val, ok = next.at(steps)
			if !ok || val != 0 {
				continue
			}
			if next == start {
				continue
			}
			// start to explore  => current step +1
			currentSteps, _ := current.at(steps)
			steps[next.i][next.j] = currentSteps + 1
			Queue = append(Queue, next)
		}
	}
	return steps
}
func main() {
	maze := readMaze("github.com/CodeDiggerSuzl/GoogleGolang/ch11/maze.in")
	for _, row := range maze {
		for _, val := range row {
			fmt.Printf("%2d", val)
		}
		fmt.Println()
	}
	fmt.Println("----------------------")
	// start point and endpoint
	steps := walk(maze, point{0, 0}, point{len(maze) - 1, len(maze[0]) - 1})
	for _, row := range steps {
		for _, val := range row {
			fmt.Printf("%3d", val)
		}
		fmt.Println()
	}
}
